SELECT
    p.title, pc.price, pc.description
FROM
    `Conciergery` c
    JOIN `ProductConciergery` pc
    ON c.id = pc.conciergery_id
    JOIN Product p
    ON pc.product_id = p.id
WHERE
    LCASE(c.title) = LCASE('6912 FOCUS LE 174') AND
    p.type='baskets' AND
    pc.endDate > NOW() AND
    REPLACE(LCASE(p.title), ' ', '') = REPLACE(LCASE('panier duo legumes'), ' ', '');
